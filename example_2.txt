Game "Coin Collector"

Define Player {
    Sprite "player.png"
    Size 50, 50
    Speed 300
    Controls WASD
}

Define Coin {
    Sprite "coin.png"
    Size 30, 30
}

Place Player at 100, 100
Place Coin at 300, 150
Place Coin at 500, 400

Rule Player touches Coin {
    AddScore 1
    PlaySound "coin.wav"
    Destroy Coin
}
