def CustomCounter(initialValue) {
    let count = State(initialValue);

    Observe(count) {
        Row(Modifier.padding(5)) {
            Text("Current Count: ");
            Text(count.get());

            Button("Increment", Modifier.padding(10)) {
                count.set(count.get() + 1);
            };
        };
    };
}

Window("Multi-Counter") {
    Column {
        CustomCounter(0);
        CustomCounter(100);
    };
};
