// 3. Classes and Inheritance
print "--- Testing OOP ---";
class Animal {
    constructor(name) {
        this.name = name;
    }

    speak() {
        print this.name + " makes a sound.";
    }
}

class Dog extends Animal {
    constructor(name) {
        super.constructor(name);
    }

    speak() {
        print this.name + " barks loudly!";
    }
}

// Note: No 'new' keyword needed in your language!
let myDog = Dog("Rex");
myDog.speak();

// 4. Reactive UI
print "--- Launching UI ---";

// Create reactive state holders
let counterState = State(0);
let textState = State("Type something...");

// Build the declarative UI tree
Window("My Custom Language UI") {
    Column(15.0) {
        Text("Welcome to my UI Framework!", 24.0, "blue");

        Row(10.0) {
            Button("Decrement") {
                counterState.set(counterState.get() - 1);
            };

            Observe(counterState) {
                Text("Count: " + counterState.get(), 18.0, "black");
            };

            Button("Increment") {
                counterState.set(counterState.get() + 1);
            };
        };

        Text("Reactivity Test:", 18.0, "gray");
        TextField(textState);

        Observe(textState) {
            Text("You typed: " + textState.get(), 14.0, "green");
        };
    };
};
