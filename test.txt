let themeState = State("Light");
let dummyState = State("Test...");

Window("Settings Panel", Modifier.background("white")) {
    // The ScrollPane wraps our column so it doesn't get cut off!
    ScrollPane(Modifier.padding(10.0)) {
        Column(Modifier.padding(10.0)) {

            Text("Select Theme:");
            // Check out the vararg options at the end!
            ComboBox(themeState, Modifier.fontSize(16.0), "Light", "Dark", "System");

            Observe(themeState) {
                Text("Current Theme: ${themeState.get()}");
            };

            // Let's generate a bunch of dummy fields to force the scrollbar to appear
            Text("Extra Settings:", Modifier.padding(20.0));
            TextField(dummyState);
            TextField(dummyState);
            TextField(dummyState);
            TextField(dummyState);
            TextField(dummyState);
            TextField(dummyState);
            TextField(dummyState);
            TextField(dummyState);
        };
    };
};
